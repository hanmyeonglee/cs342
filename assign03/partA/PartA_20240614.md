## A-1. CNF, Q1

$$
\begin{align*}
\varphi:&=\neg(p\rightarrow q)\vee((r\vee s)\rightarrow(q\vee t))\vee(\neg p\rightarrow\neg v)\\
&=\neg(\neg p\vee q)\vee(\neg(r\vee s)\vee(q\vee t))\vee(\neg\neg p\vee\neg v)&\text{ by implication elimination}\\
&=(\neg\neg p\wedge\neg q)\vee((\neg r\wedge\neg s)\vee(q\vee t))\vee(\neg\neg p\vee\neg v)&\text{ by de Morgan}\\
&=(p\wedge\neg q)\vee((\neg r\wedge\neg s)\vee(q\vee t))\vee(p\vee\neg v)&\text{ by double-negation elimination}\\
&=(p\wedge\neg q)\vee(\neg r\wedge\neg s)\vee q\vee t\vee p\vee\neg v&\text{ by associativity}\\
&=p\vee(p\wedge\neg q)\vee(\neg r\wedge\neg s)\vee q\vee t\vee\neg v&\text{ by commutativity}\\
&=p\vee(\neg r\wedge\neg s)\vee q\vee t\vee\neg v&\text{ by absorption law}\\
&=p\vee q\vee t\vee\neg v\vee(\neg r\wedge\neg s)&\text{ by commutativity}\\
&=(p\vee q\vee t\vee\neg v)\vee(\neg r\wedge\neg s)&\text{ by associativity}\\
&=(p\vee q\vee t\vee\neg v\vee\neg r)\wedge(p\vee q\vee t\vee\neg v\vee\neg s)&\text{ by associativity}\\
\end{align*}
$$
